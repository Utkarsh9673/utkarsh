        // --- Three.js Background Animation ---
        let scene, camera, renderer, lines;
        const canvasContainer = document.getElementById('threejs-canvas');

        function initThreeJS() {
            // Check if container exists
            if (!canvasContainer) {
                console.error("Three.js canvas container not found.");
                return;
            }

            try {
                scene = new THREE.Scene();
                camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 1, 1000);
                camera.position.z = 5;

                renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
                renderer.setSize(window.innerWidth, window.innerHeight);
                renderer.setPixelRatio(window.devicePixelRatio);
                renderer.setClearColor(0x000000, 0);
                canvasContainer.appendChild(renderer.domElement);

                // Get the NEW accent color from CSS variables
                const accentColorCSS = getComputedStyle(document.documentElement).getPropertyValue('--accent-color').trim();
                const gridColor = new THREE.Color(accentColorCSS);


                 const material = new THREE.LineBasicMaterial({
                     color: gridColor,
                     transparent: true,
                     opacity: 0.35 // Slightly increased opacity for the darker theme
                 });

                 const size = 50;
                 const divisions = 40;
                 const points = [];
                 const step = size / divisions;
                 const halfSize = size / 2;

                 for (let i = 0; i <= divisions; i++) {
                     points.push(new THREE.Vector3(-halfSize + i * step, -halfSize, 0));
                     points.push(new THREE.Vector3(-halfSize + i * step, halfSize, 0));
                     points.push(new THREE.Vector3(-halfSize, -halfSize + i * step, 0));
                     points.push(new THREE.Vector3(halfSize, -halfSize + i * step, 0));
                 }

                 const geometry = new THREE.BufferGeometry().setFromPoints(points);
                 lines = new THREE.LineSegments(geometry, material);
                 lines.rotation.x = Math.PI / 4.5; // Adjusted tilt slightly
                 scene.add(lines);


                window.addEventListener('resize', onWindowResize, false);
                animateThreeJS();

             } catch (error) {
                console.error("Error initializing Three.js:", error);
                // Optionally hide the canvas or show a fallback
                if (canvasContainer) canvasContainer.style.display = 'none';
            }
        }

        function onWindowResize() {
            if (!camera || !renderer) return; // Add check
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        let mouseX = 0, mouseY = 0;
        const mouseSensitivity = 0.08; // Slightly reduced sensitivity
        document.addEventListener('mousemove', (event) => {
             mouseX = ((event.clientX / window.innerWidth) * 2 - 1) * mouseSensitivity;
             mouseY = (-(event.clientY / window.innerHeight) * 2 + 1) * mouseSensitivity;
         });


        function animateThreeJS() {
             if (!renderer || !scene || !camera || !lines) { // Add check
                 requestAnimationFrame(animateThreeJS); // Still request next frame
                 return;
             }
             requestAnimationFrame(animateThreeJS);
             const time = Date.now() * 0.00008; // Slowed down rotation speed

             lines.rotation.z = time * 0.4;
             // Make mouse follow effect smoother/slower
             lines.rotation.y += (mouseX - lines.rotation.y) * 0.015;
             lines.rotation.x += (-mouseY - lines.rotation.x + Math.PI / 4.5) * 0.015; // Match initial tilt


             renderer.render(scene, camera);
        }

        // Wait for DOM content and CSS to potentially load before init
         if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', initThreeJS);
         } else {
            initThreeJS(); // DOMContentLoaded has already fired
         }


    </script>
